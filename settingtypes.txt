
[General]

#   Allow 1042 to automaticly adjust your settings to what the game requires.
#
#   Recomended for novice players
1042_auto_adjust_settings (Automaticly adjust settings) bool false

#   Hardcore Mode
#
#   Not recomended for novice players
1042_enable_hardcore (Enable Hardcore Mode) bool false

#   1042 Lore Language for items, nodes, etc.
#
#   This changes only 1042 text and is not a translation
1042_lorelang (Enable Lore Language) bool false

#   Set camera to Third Person (Back) when gliding
#
#   Whether to update the camera during gliding.
1042_flight_cam (Update Camera During Flight) bool true


[Server settings]

#   Warn players to adjust settings on join.
1042_warn_players_about_settings (Warn players about settings) bool true


[Mapgen]

#   Enable mapgen decorations
#
#   Recomended: true
1042_mapgen_decorations (Enable mapgen decorations) bool true


[Advanced]

#   Ignore game required settings, and force run.
#
#   ***THIS IS NOT RECOMENDED***
1042_ignore_required_settings (Force override required settings) bool false

#   Disable in-game weather
#
#   ***THIS IS NOT RECOMENDED***
1042_disable_weather (Disable in-game weather) bool false


[Testing/Debug]

#   Enable and run the intigrated tests mod (WIP)
#
#   ***THIS IS NOT RECOMENDED***
1042_enable_intigrated_tests (Enable intigrated tests) bool false

#   Enable experimental Movement Physics
1042_experimental_physics (Enable Experimental Physics) bool false
